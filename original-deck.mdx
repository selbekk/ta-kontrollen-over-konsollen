## Gruppering
```js
console.group('action');
console.info('previous state', state);
console.info('action payload', action);
console.info('next state', reducer(state, action));
console.groupEnd('action');
```

<Button onClick={() => {
  const action = { type: 'SAY_HI', name: 'Kris' };
  const state = { salute: '' };
  const reducer = (prevState, theAction) => {
    if (theAction.type === 'SAY_HI') {
      return { salute: `Hi ${theAction.name}` };
    }
    return state;
  };
  console.group('action');
  console.info('previous state', state);
  console.info('action payload', action);
  console.info('next state', reducer(state, action));
  console.groupEnd('action');
}}>Vis meg!</Button>
